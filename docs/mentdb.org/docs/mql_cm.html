Description

    Set a connection

Parameters

    key:   The key - string - required
    object:   The JSON object - string - required
admin
cm set "demo_cm_mentdb" {execute "mentdb.remote.config.get"
	"[hostname]" "localhost"
	"[port]" "9998"
	"[key]" "pwd"
	"[user]" "admin"
	"[password]" "pwd"
	"[connectTimeout]" "10000"
	"[readTimeout]" "30000"
	"[subTunnels]" (mql {

tunnel execute_hot "tunnelId1" {cm get "demo_cm_mentdb";} (mql {
	tunnel execute_hot "tunnelId2" {cm get "demo_cm_mentdb";} (mql {
	
		[MQL_TO_REPLACE]
		
	});
});

})
;};
