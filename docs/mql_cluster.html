cluster execute_hot

Description

    Execute a MQL command in hot throught a cluster.

admin
-> "[local_var1]" "data1";
-> "[local_var2]" "data2";
include "cluster.1n.hot.exe"
	"[cluster_id]" "cluster_id_1"
	"[method]" "LB_50_50|SIGNAL"
	"[tunnel_id]" "tunnel_id_1"
	"[request]" (concat 
		"-> \"[remote_var1]\" " (mql encode [local_var1]) ";"
		"-> \"[remote_var2]\" " (mql encode [local_var2]) ";"
		(mql {
			concat [remote_var1] ":" [remote_var2]
		})
	)
;
mentdb
data1:data2
